```typescript
function twoSum(nums: number[], target: number): number[] {
  const resMap: Map<number, number> = new Map;
  let j: number | undefined = 0;
  for (let i = 0; i < nums.length; i++) {
    j = resMap.get(target - nums[i])
    if (j !== undefined)
      return [j, i];
    else
      resMap.set(nums[i], i)
  }
  return [];
};
```
**Note:**
Use map to ensure the search of O(1);
